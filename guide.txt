الخطوة 5: إعداد Rclone على EC2

s) Set configuration password
q) Quit config

# اختر n
n

# Enter name for new remote:
name> gdrive_source

# Type of storage to configure:
Storage> drive

# Google Application Client Id
client_id> [اضغط Enter - اتركها فاضية]

# OAuth Client Secret
client_secret> [اضغط Enter]

# Scope that rclone should use
scope> 1  [Full access]

# ID of the root folder

# Service Account Credentials JSON file path
service_account_file> [اضغط Enter]

# Edit advanced config?
y/n> n

# Use web browser to automatically authenticate rclone with remote?
y/n> n  [لأننا على سيرفر بدون GUI]

📝 Script كامل للتنفيذ السريع
احفظ هذا في ملف demo_migration.sh:
bash
#!/bin/bash

# Demo Migration Script
# Run on AWS EC2

set -e

echo "🚀 Starting Cloud Migration Demo..."

# Colors
GREEN='\033[0;32m'
YELLOW='\033[1;33m'
NC='\033[0m'

# Install Rclone
echo -e "${YELLOW}Installing Rclone...${NC}"
curl https://rclone.org/install.sh | sudo bash

# Create directories
mkdir -p ~/demo_logs

echo -e "${GREEN}✓ Rclone installed${NC}"
echo ""
echo "Next steps:"
echo "1. Run: rclone config"
echo "2. Configure source account (gdrive_source)"
echo "3. Configure destination account (gdrive_destination)"
echo "4. Test: rclone lsd gdrive_source:"
echo "5. Run migration: ./run_migration.sh"
ملف run_migration.sh:
bash
#!/bin/bash

# Migration execution script

TIMESTAMP=$(date +%Y%m%d_%H%M%S)
LOG_FILE="~/demo_logs/migration_${TIMESTAMP}.log"

echo "Starting migration at $(date)"

# Dry run first
echo "Running dry-run..."
rclone copy gdrive_source: gdrive_destination:Migration_Demo_${TIMESTAMP} \
  --dry-run \
  --progress \
  --log-file="${LOG_FILE}.dryrun"

echo "Dry-run complete. Check log: ${LOG_FILE}.dryrun"
echo "Press Enter to continue with actual migration, or Ctrl+C to cancel"
read

# Actual migration
echo "Starting actual migration..."
rclone copy gdrive_source: gdrive_destination:Migration_Demo_${TIMESTAMP} \
  --progress \
  --transfers 4 \
  --checkers 8 \
  --stats 10s \
  --log-file="${LOG_FILE}"

echo "Migration complete!"

# Verification
echo "Running verification..."
rclone check gdrive_source: gdrive_destination:Migration_Demo_${TIMESTAMP} \
  --one-way \
  --log-file="${LOG_FILE}.verify"

echo "✓ Migration and verification complete!"
echo "Logs saved to: ${LOG_FILE}"
